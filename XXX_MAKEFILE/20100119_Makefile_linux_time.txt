## use -O, -O1, -O2 for optimization


SOURCE = name.c
OBJECTS = $(SOURCE:%.c=%.o)
CC = gcc
CFLAGS += -g -Wall
RM = rm -f

%.o: %.c
	$(CC) -c $(CFLAGS) $<


all: name.exe

name.exe: $(OBJECTS)
	$(CC) -o $@ $(OBJECTS) -lm -lrt
# linked against librt for the time.h functions -> -lrt



clean:
	$(RM) $(OBJECTS) name.exe *~

.PHONY: clean all
