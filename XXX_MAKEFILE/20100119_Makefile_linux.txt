# Makefile (simple)
#
# use -O, -O1, -O2 for optimization 


SOURCE = name.c
OBJECTS = $(SOURCE:%.c=%.o)
CC = gcc
CFLAGS += -g -Wall -std=c99 -pedantic
RM = rm -f

%.o: %.c
	$(CC) -c $(CFLAGS) $<

name.exe: $(OBJECTS)
	$(CC) -o $@ $(OBJECTS) -lm

all: name.exe

clean: 
	$(RM) $(OBJECTS) name.exe *~

.PHONY: clean all
