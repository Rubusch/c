## use -O, -O1, -O2 for optimization


SOURCE = name.c
OBJECTS = $(SOURCE:%.c=%.o)
CC = gcc
CFLAGS += -g -Wall -std=c99 -pedantic
RM = rm -f

%.o: %.c
	$(CC) -c $(CFLAGS) $<


all: name.exe

name.exe: $(OBJECTS)
	$(CC) -o $@ $(OBJECTS) -lm -D _SVID_SOURCE=1 -D _XOPEN_SOURCE=600


clean:
	$(RM) $(OBJECTS) name.exe *~


.PHONY: all clean
